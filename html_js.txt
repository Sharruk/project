🔥 COMPLETE ATM SYSTEM (SIMPLE, ONE FILE)

You can literally write something like this in exam (or simplified parts of it).

<!DOCTYPE html>
<html>
<body>

<h2>ATM System</h2>

<!-- LOGIN SECTION -->
<div id="login">

<h3>Login</h3>

PIN: <input type="password" id="pin"><br><br>

<button onclick="login()">Login</button>

<p id="loginMsg"></p>

</div>


<!-- MENU SECTION -->
<div id="menu" style="display:none;">

<h3>ATM Menu</h3>

<select id="option">
<option>Withdraw</option>
<option>Deposit</option>
<option>Balance</option>
</select>

<br><br>

Amount: <input id="amount"><br><br>

<button onclick="process()">Submit</button>

<p id="output"></p>

</div>


<script>

// default values
var correctPIN = 1234;
var balance = 5000;

// LOGIN FUNCTION
function login() {

var enteredPIN = document.getElementById("pin").value;

if (enteredPIN == correctPIN) {

document.getElementById("login").style.display = "none";
document.getElementById("menu").style.display = "block";

}
else {

document.getElementById("loginMsg").innerHTML = "Wrong PIN";

}

}


// ATM OPERATIONS
function process() {

var option = document.getElementById("option").value;
var amt = Number(document.getElementById("amount").value);

// WITHDRAW
if (option == "Withdraw") {

if (amt % 100 != 0) {

document.getElementById("output").innerHTML = "Enter multiples of 100";

}
else if (amt > balance) {

document.getElementById("output").innerHTML = "Insufficient Balance";

}
else {

balance = balance - amt;
document.getElementById("output").innerHTML = "Withdraw Success. Balance = " + balance;

}

}

// DEPOSIT
else if (option == "Deposit") {

balance = balance + amt;
document.getElementById("output").innerHTML = "Deposit Success. Balance = " + balance;

}

// BALANCE CHECK
else if (option == "Balance") {

document.getElementById("output").innerHTML = "Current Balance = " + balance;

}

}

</script>

</body>
</html>
🧠 What this shows (important for marks)

This single file demonstrates:

✔ Login validation (PIN check)
✔ Menu selection
✔ Withdraw logic (multiple of 100 + balance check)
✔ Deposit logic
✔ Balance enquiry
✔ Input + processing + output
